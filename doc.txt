
Sub UpdateFormula()
    Dim rng As Range
    Dim cell As Range
    
    ' Select the column range dynamically
    Set rng = Selection
    Set rng = Range(rng, rng.End(xlDown))
    
    ' Loop through each cell in the column range
    For Each cell In rng
        ' Check if the cell value is numeric
        If Not IsNumeric(cell.Value) Then
            ' Update the formula if the cell value is non-numeric
            cell.Formula = Replace(cell.Formula, "$G$1", """xyz""")
        End If
    Next cell
End Sub
