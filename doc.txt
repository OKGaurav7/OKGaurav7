# Apply conditional formatting to column "Trade Price"
trade_price_column = None
for cell in ws[1]:
    if cell.value == 'Trade Price':
        trade_price_column = cell.column
        break

if trade_price_column:
    ws.column_dimensions[get_column_letter(trade_price_column)].width = 10  # Adjust column width to 10
    for row in ws.iter_rows(min_row=2):
        for cell in row:
            if cell.column == trade_price_column:
                if isinstance(cell.value, (int, float)) and cell.value > 0:
                    cell.fill = PatternFill(start_color="FFC7CE", end_color="FFC7CE", fill_type="solid")
