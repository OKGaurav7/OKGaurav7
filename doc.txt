import pandas as pd

# Assuming df is your DataFrame
# Assuming 'existing_sheet.xlsx' is your existing Excel sheet

# Load the existing Excel file
existing_excel = pd.ExcelWriter('existing_sheet.xlsx', engine='openpyxl')
existing_df = pd.read_excel(existing_excel, sheet_name='Sheet1')

# Assuming new_df is the DataFrame you want to write
new_df = pd.DataFrame({'data': [1, 2, 3]})  # Example DataFrame

# Write the new DataFrame to the existing Excel sheet starting from column 'U'
new_df.to_excel(existing_excel, sheet_name='Sheet1', startcol=20, index=False, header=False)

# Save changes
existing_excel.save()
