Sub CheckNumericValues()
    Dim rng As Range
    Dim cell As Range
    
    ' Specify the column range you want to check
    Set rng = Range("A:A")  ' Change "A:A" to your desired column range
    
    ' Loop through each cell in the column range
    For Each cell In rng
        ' Check if the cell value is numeric
        If IsNumeric(cell.Value) Then
            ' Cell value is numeric
            cell.Interior.Color = RGB(0, 255, 0)  ' Change cell background color to green
        Else
            ' Cell value is not numeric
            cell.Interior.Color = RGB(255, 0, 0)  ' Change cell background color to red
        End If
    Next cell
End Sub
